    <!-- APlayer 容器 -->

    <!-- 引入 APlayer 和 MetingJS -->
    <script src="https://cdn.jsdelivr.net/npm/aplayer/dist/APlayer.min.js"></script>
    <script src="https://cdn.jsdelivr.net/npm/meting@2/dist/Meting.min.js"></script>

    <script>
        // 全局变量用于存储APlayer实例
        let ap = null;

        // 初始化播放器
        function initPlayer() {
            // 等待MetingJS加载完成
            const checkInterval = setInterval(() => {
                const playerElement = document.getElementById('aplayer');
                if (playerElement && playerElement.aplayer) {
                    clearInterval(checkInterval);
                    ap = playerElement.aplayer;
                    console.log('APlayer实例已获取', ap);
                    
                    // 恢复播放状态
                    restorePlaybackState();
                }
            }, 100);
        }

        // 恢复播放状态
        function restorePlaybackState() {
            const playInfo = JSON.parse(localStorage.getItem("playInfo"));
            if (!playInfo) {
                console.log('没有找到之前的播放记录');
                return;
            }

            // 等待APlayer完全加载
            setTimeout(() => {
                try {
                    // 切换歌曲
                    ap.list.switch(playInfo.index);
                    console.log(`正在恢复第${playInfo.index + 1}首歌曲...`);
                    
                    // 等待歌曲加载
                    setTimeout(() => {
                        // 调整播放进度
                        ap.seek(playInfo.currentTime);
                        
                        // 恢复播放/暂停状态
                        if (!playInfo.paused) {
                            ap.play();
                            console.log('已恢复播放');
                        } else {
                            console.log('已恢复播放进度（暂停状态）');
                        }
                    }, 800);
                } catch (error) {
                    console.error('恢复播放状态时出错:', error);
                }
            }, 500);
        }

        // 页面销毁前监听
        window.onbeforeunload = () => {
            if (ap) {
                // 将播放信息用对象封装，并存入到localStorage中
                const playInfo = {
                    index: ap.list.index,
                    currentTime: ap.audio.currentTime,
                    paused: ap.audio.paused
                };
                localStorage.setItem("playInfo", JSON.stringify(playInfo));
                console.log('播放状态已保存', playInfo);
            }
        };

        // 页面加载后初始化播放器
        window.addEventListener('load', initPlayer);
    </script>
</body>

<!--改为了本地引入 <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/aplayer/dist/APlayer.min.css"> -->
<script src="https://cdn.jsdelivr.net/npm/aplayer/dist/APlayer.min.js"></script>

<div id="aplayer"></div>

<script>
	// 音乐数据预处理
	const musicDir = '{{ .Site.Home.Permalink }}music/';
	const songs = [
		'任贤齐-伤心太平洋',
		'张韶涵-隐形的翅膀'
	];
	
	let audios = [];
	songs.forEach((item) => {
		const songInfo = item.split('-');
		audios.push({
			name: songInfo[1],
			artist: songInfo[0],
			url: musicDir + item + '/song.mp3', // 歌曲路径
			cover: musicDir + item + '/cover.jpg', // 封面路径
			lrc: musicDir + item + '/lyric.lrc', // 歌词路径
		});
	});
	
	const ap = new APlayer({
		container: document.getElementById('aplayer'),
		fixed: true,
		lrcType: 3,
		audio: audios
	});

	 /**
     * 页面销毁前监听
     */
    window.onbeforeunload = () => {
        // 将播放信息用对象封装，并存入到localStorage中
        const playInfo = {
            index: ap.list.index,
            currentTime: ap.audio.currentTime,
            paused: ap.paused
        };
        localStorage.setItem("playInfo", JSON.stringify(playInfo));
    };

    /**
     * 页面加载后监听
     */
    window.onload = () => {
        // 从localStorage取出播放信息
        const playInfo = JSON.parse(localStorage.getItem("playInfo"));
        if (!playInfo) {
            return;
        }
        // 切换歌曲
        ap.list.switch(playInfo.index);
        // 等待500ms再执行下一步(切换歌曲需要点时间，不能立马调歌曲进度条)
        setTimeout(() => {
            // 调整时长
            ap.seek(playInfo.currentTime);
            // 是否播放
            if (!playInfo.paused) {
                ap.play()
            }
        }, 500);
    };
</script>

    <script>
        // 全局变量用于存储APlayer实例
        let ap = null;
        let currentPlaylistId = '117882154';

        // 初始化播放器
        function initPlayer() {
            // 等待MetingJS加载完成
            const checkInterval = setInterval(() => {
                const playerElement = document.getElementById('aplayer');
                if (playerElement && playerElement.aplayer) {
                    clearInterval(checkInterval);
                    ap = playerElement.aplayer;
                    console.log('APlayer实例已获取', ap);
                    
                    // 恢复播放状态
                    restorePlaybackState();
                }
            }, 100);
        }

        // 恢复播放状态
        function restorePlaybackState() {
            const playInfo = JSON.parse(localStorage.getItem("playInfo"));
            if (!playInfo) {
                console.log('没有找到之前的播放记录');
                return;
            }

            // 等待APlayer完全加载
            setTimeout(() => {
                try {
                    // 切换歌曲
                    ap.list.switch(playInfo.index);
                    console.log(`正在恢复第${playInfo.index + 1}首歌曲...`);
                    
                    // 等待歌曲加载
                    setTimeout(() => {
                        // 调整播放进度
                        ap.seek(playInfo.currentTime);
                        
                        // 恢复播放/暂停状态
                        if (!playInfo.paused) {
                            ap.play();
                            console.log('已恢复播放');
                        } else {
                            console.log('已恢复播放进度（暂停状态）');
                        }
                    }, 800);
                } catch (error) {
                    console.error('恢复播放状态时出错:', error);
                }
            }, 500);
        }

        // 页面销毁前监听
        window.onbeforeunload = () => {
            if (ap) {
                // 将播放信息用对象封装，并存入到localStorage中
                const playInfo = {
                    index: ap.list.index,
                    currentTime: ap.audio.currentTime,
                    paused: ap.audio.paused
                };
                localStorage.setItem("playInfo", JSON.stringify(playInfo));
                console.log('播放状态已保存', playInfo);
            }
        };

        // 创建Meting元素
        function createMetingElement(playlistId) {
            const existingMeting = document.querySelector('meting-js');
            if (existingMeting) existingMeting.remove();
            
            const metingElement = document.createElement('meting-js');
            metingElement.setAttribute('server', 'netease');
            metingElement.setAttribute('type', 'playlist');
            metingElement.setAttribute('id', playlistId);
            metingElement.setAttribute('fixed', 'true');
            metingElement.setAttribute('autoplay', 'false');
            
            document.getElementById('aplayer').appendChild(metingElement);
            
            // 重新初始化播放器
            initPlayer();
        }

        // 按钮点击事件
        document.querySelectorAll('button').forEach(button => {
            button.addEventListener('click', function() {
                const playlistId = this.getAttribute('data-id');
                
                // 更新按钮状态
                document.querySelectorAll('button').forEach(btn => {
                    btn.classList.remove('active');
                });
                this.classList.add('active');
                
                // 保存当前播放状态
                if (ap) {
                    const playInfo = {
                        index: ap.list.index,
                        currentTime: ap.audio.currentTime,
                        paused: ap.audio.paused
                    };
                    localStorage.setItem("playInfo", JSON.stringify(playInfo));
                }
                
                // 创建新的Meting元素
                createMetingElement(playlistId);
                currentPlaylistId = playlistId;
            });
        });

        // 初始创建Meting元素
        createMetingElement(currentPlaylistId);
    </script>
